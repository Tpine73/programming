문제

생일과 오늘을 입력하면 태어난지 몇 일째인지 출력하는 알고리즘입니다 이를 해결하는 다음 프로그램을 해석하세요 

(Hint) 1년1월1일부터 해당일까지 전체 날짜 계산하는 방법 
		
 
			

				   1년1월1일 ~ 직전년도 12월31일   
				
				   해당년도 1월1일 ~ 직전달 말일   
				
				   해당달 1일 ~ 해당일

프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter05/Date.c 

	#include <stdio.h>  

 

	// 프로그램 시작 

1	int main() 

	{ 

		// 월별 말일 및 임시변수 초기화 

		int lastDayPerMonth[13] = { 0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };  

		int previousYear  = 0; 

		int previousMonth = 0; 

	 

		// 생일 날짜 초기화 

		int birthYear  = 0; 

		int birthMonth  = 0; 

		int birthDay = 0; 

		int birthTotalDays = 0; 

	 

		// 오늘 날짜 초기화 

		int thisYear  = 0; 

		int thisMonth  = 0; 

		int thisDay = 0; 

		int thisTotalDays = 0; 

	 

		// 살아온 기간 초기화 

		int duration = 0; 

	 

		// 생일 날짜를 입력 

		// 2019년 1월1일을 입력한다고 가정 

		printf( "생년을 입력하세요: " ); 

		scanf( "%d", &birthYear ); 

		printf( "생월을 입력하세요: " ); 

		scanf( "%d", &birthMonth ); 

		printf( "생일을 입력하세요: " ); 

2		scanf( "%d", &birthDay ); 

	 

		// 오늘 날짜를 입력 

		// 2019년 2월 1일을 입력한다고 가정 

		printf( "올해를 입력하세요: " ); 

		scanf( "%d", &thisYear ); 

		printf( "이번 달을 입력하세요: " ); 

		scanf( "%d", &thisMonth ); 

		printf( "오늘 날짜를 입력하세요: " ); 

3		scanf( "%d", &thisDay ); 

	 

		// 살아온 기간 계산 

		// 1년1월1일부터 생일 날짜까지 전체 날짜 계산 

		// 1년1월1일부터 직전년도 12월31일까지 날짜 계산 

		previousYear = birthYear - 1; 

4		birthTotalDays = previousYear * 365 + previousYear / 4 - previousYear / 100 + previousYear / 400; 

		// 해당년도 1월1일부터 직전달 말일까지 날짜 계산 

5    		for ( previousMonth = 1; previousMonth < birthMonth; previousMonth++ )  

		{ 

6			birthTotalDays += lastDayPerMonth[ previousMonth ];						 

			// 윤년 2월의 경우 2월29일을 반영  

7			if ( ( previousMonth == 2 )  

				&& ( ( ( birthYear % 400 ) == 0 )  

				    || ( ( ( birthYear % 100 ) != 0 ) && ( ( birthYear % 4 ) == 0 ) ) ) ) 

			{ 

8				birthTotalDays++; 

			} 

		} 

		// 해당달 1일부터 해당일까지 날짜 계산 

		birthTotalDays += birthDay; 

	 

		// 1년1월1일부터 오늘 날짜까지 전체 날짜 계산 

		// 1년1월1일부터 직전년도 12월31일까지 날짜 계산 

		previousYear = thisYear - 1; 

9		thisTotalDays = previousYear * 365 +  previousYear / 4 - previousYear / 100 + previousYear / 400; 

		// 해당년도 1월1일부터 직전달 말일까지 날짜 계산 

10    		for ( previousMonth = 1; previousMonth < thisMonth; previousMonth++ )  

		{ 

11			thisTotalDays += lastDayPerMonth[ previousMonth ];						 

			// 윤년 2월의 경우 2월29일을 반영  

12			if ( ( previousMonth == 2 )  

				&& ( ( ( thisYear % 400 ) == 0 )  

				    || ( ( ( thisYear % 100 ) != 0 ) && ( ( thisYear % 4 ) == 0 ) ) ) ) 

			{ 

13				thisTotalDays++; 

			} 

		} 

	 

		// 해당달 1일부터 해당일까지 날짜 계산 

		thisTotalDays += thisDay; 

	 

		// 살아온 기간은 생일부터 오늘까지 전체 날짜의 차이로 계산 

		duration = thisTotalDays - birthTotalDays + 1; 

	 

		// 몇 일째 살아왔는지 출력 

14		printf( "당신은 %d년 %d월 %d일부터 %d년 %d월 %d일까지 태어난지 %d일째군요",  

				birthYear, birthMonth, birthDay, thisYear, thisMonth, thisDay, duration ); 

	 

		// 프로그램 종료 

15		return 0; 

	} 

		
		

	
	

	

		
키워드 설명

		

int 
  -2, -1, 0, 1, 2 등의 정수 자료형


{ 
  여는 중괄호로 배열을 선언하면서 초기화할 때 사용


} 
  닫는 중괄호로 배열을 선언하면서 초기화할 때 사용


/ 
  나눗셈 연산자


4 
  4로 나누어 떨어지는 해는 윤년을 의미


100 
  100으로 나누어 떨어지는 해는 평년을 의미


400 
  400으로 나누어 떨어지는 해는 윤년을 의미


+= 
  복합대입연산자로 a += b는 a = a + b를 의미하며 변수공간a  a값 을 변수공간a  a값 + b값 으로 변경


% 
  나머지 연산자로 정수 a, b에 대해 a % b는 a를 b로 나누었을 때의 나머지를 연산하며, 참고로 몫은 a / b로 연산
 예를 들어 14 % 4 = 2 이고 14 / 4 = 3
