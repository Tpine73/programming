프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter05/LittleStar.c 

	#include <stdio.h> 					 

	#include <unistd.h> 

 

	// 프로그램시작 

1	int main() 

	{ 

		int index = 0; 

		char lyrics [16][6] = { "반", "짝", " 반", "짝", " 작", "은", " 별", " ", "아", "름", "답", "게", " 비", "치", "네", "\0" }; 

 

		// 먼저 노란 별 모양을 출력 

2		printf(" \033[33m \n" ); 

3		printf("              &@ \n"); 

4		printf("             (@@@. \n"); 

5		printf("            #@@@@@ \n"); 

6		printf("           /@@@@@@@ \n"); 

7		printf("  @@@@@@@@@@@@@@@@@@@@@@@@@@@ \n"); 

8		printf("    /@@@@@@@@@@@@@@@@@@@@@. \n"); 

9		printf("       @@@@@@@@@@@@@@@@ \n"); 

10		printf("        #@@@@@@@@@@@@@ \n"); 

11		printf("        @@@@@@@@@@@@@@@ \n"); 

12		printf("       @@@@@@@   @@@@@@ \n"); 

13		printf("       @@@           @@@ \n"); 

14		printf("       @               @ \033[0m \n"); 

		 

		// 작은별 노래를 박자에 맞춰 출력 

15		for ( index = 0; lyrics[ index ][0] != '\0'; index++ )

16			printf( "%s", lyrics[ index ] ); 

17			fflush( stdout );

		}

키워드 설명

		

[16][6] 
  6칸씩 16줄로 구성된 2차원 배열을 선언


[ index ] 
  배열의 여러 칸 중에서 index번째 칸을 의미하며, 참고로 배열은 0번째 칸부터 시작


'\0' 
  빈문자(NULL문자)로 \0는 문자열 끝을 나타냄


%s 
  문자열  형식으로 입출력


fflush 
  입출력 효율성을 고려하여, 일반적으로 입출력 요청이 있을때마다 처리하지 않고, 입출력 버퍼가 쌓일 때까지 기다렸다고 한꺼번에 처리하는데, fflush는 입출력 버퍼가 쌓일 때까지 기다리지 않고 강제로 처리


stdout 
  모니터 등의 표준 출력


sleep 
  지정한 시간만큼 대기하는 함수
리눅스/유닉스 버전 sleep(초)
#include <stdio.h> 					
#include <unistd.h>
int main()
{
	printf( "3초 대기 시작\n");
	sleep(3);
	printf( "3초 대기 종료\n");
	return 0;
}
윈도우 버전 Sleep(밀리초)
#include <stdio.h> 					
#include <Windows.h>
int main()
{
	printf( "3초 대기 시작\n");
	Sleep(3000);
	printf( "3초 대기 종료\n");
	return 0;
}
