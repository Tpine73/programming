키워드 설명

		

include 
  헤더파일(.h)을 소스코드(.c)에 포함하는 명령문


stdio.h 
  standard input output의 줄임말로 printf나 scanf 등 표준입출력 관련 함수 정보를 가지고 있는 헤더파일


unistd.h 
  sleep 함수를 포함하는 헤더 파일


main 
  프로그램을 시작할 때 운영체제가 호출하는 함수


printf 
  print formatted의 줄임말로 전달받은 내용을 형식에 맞게 모니터에 출력하는 함수


\033 
  출력 설정 이스케이프 문자 
printf( "\033[31m 글자색 빨강 \033[0m\n" ); // 출력결과: 글자색 빨강 
printf( "\033[33m 글자색 노랑 \033[0m\n" ); // 출력결과:  글자색 노랑 
printf( "\033[32m 글자색 초록 \033[0m\n" ); // 출력결과:  글자색 초록 
printf( "\033[34m 글자색 파랑 \033[0m\n" ); // 출력결과:  글자색 파랑 
printf( "\033[41m 바탕색 빨강 \033[0m\n" ); // 출력결과:  바탕색 빨강  
printf( "\033[43m 바탕색 노랑 \033[0m\n" ); // 출력결과:  바탕색 노랑 
printf( "\033[42m 바탕색 초록 \033[0m\n" ); // 출력결과:  바탕색 초록 
printf( "\033[44m 바탕색 파랑 \033[0m\n" ); // 출력결과:  바탕색 파랑 
printf( "\033[01m 글자 굵게    \033[0m\n" );   // 출력결과:  글자 굵게 
 ※ \033[0m는 글자색이나 배경색 초기화
 ※ \033[H\033[J는 모니터 화면을 깨끗이 지우고 초기화


\n 
  줄바꿈(new line) 문자


sleep 
  지정한 시간만큼 대기하는 함수
리눅스/유닉스 버전 sleep(초)
#include <stdio.h> 					
#include <unistd.h>
int main()
{
	printf( "3초 대기 시작\n");
	sleep(3);
	printf( "3초 대기 종료\n");
	return 0;
}
윈도우 버전 Sleep(밀리초)
#include <stdio.h> 					
#include <Windows.h>
int main()
{
	printf( "3초 대기 시작\n");
	Sleep(3000);
	printf( "3초 대기 종료\n");
	return 0;
}


return 
  함수 종료시 실행흐름이 함수를 호출했던 곳으로 되돌아감 
예를 들어, main 함수가 끝나면 main 함수를 호출한 운영체제로 제어흐름이 돌아가고 프로그램이 종료
