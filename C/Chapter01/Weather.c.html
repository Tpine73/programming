프로그램 코드 

	// 파일명 : ./Chapter01/Weather.c 

	#include <stdio.h>											 

	 

	// 프로그램 시작 

1	int main() 

	{ 

		// 날씨 내용을 저장하는 변수 초기화 

		char month[16] = ""; 

		char day[16] = ""; 

		char week[16] = ""; 

		char weatherReport[16] = ""; 

		char precaution[16] = ""; 

		int  lowSeoul = 0; 

		int  lowChuncheon = 0; 

		int  lowDaejeon = 0; 

		int  lowGwangju = 0; 

		int  lowBusan = 0; 

		int  lowJeju = 0; 

		int  highSeoul = 0; 

		int  highChuncheon = 0; 

		int  highDaejeon = 0; 

		int  highGwangju = 0; 

		int  highBusan = 0; 

2		int  highJeju = 0; 

		 

		// 주요 날씨 내용을 입력 

		// 오늘날짜, 기상특보, 주의사항을 입력 

3		printf( "오늘은 몇 월인가요: " );								 

		// 1월을 입력한다고 가정 

4		scanf( "%s", month );										 

5		printf( "오늘은 몇 일인가요: " );								 

		// 31일을 입력한다고 가정 

6		scanf( "%s", day );										 

7		printf( "오늘은 무슨 요일인가요: " );								 

		// 목요일을 입력한다고 가정 

8		scanf( "%s", week );										 

9		printf( "오늘의 기상특보는 무엇인가요: " );							 

		// 대설특보를 입력한다고 가정 

10		scanf( "%s", weatherReport );								 

11		printf( "오늘의 주의사항은 무엇인가요: " );							 

		// 눈길을 입력한다고 가정 

12		scanf( "%s", precaution );									 

	 

		// 지역별 최저 기온 정보를 입력 

		// 서울, 춘천, 대전, 광주, 부산, 제주의 최저 기온을 입력 

13		printf( "\n서울 최저기온은 몇 도인가요: " );							 

		// -4도를 입력한다고 가정 

14		scanf( "%d", &lowSeoul );								 

15		printf( "춘천 최저기온은 몇 도인가요: " );							 

		// -4도를 입력한다고 가정 

16		scanf( "%d", &lowChuncheon );							 

17		printf( "대전 최저기온은 몇 도인가요: " );							 

		// -1도를 입력한다고 가정		 

18		scanf( "%d", &lowDaejeon );							 

19		printf( "광주 최저기온은 몇 도인가요: " );							 

		// 0도를 입력한다고 가정 

20		scanf( "%d", &lowGwangju ); 							 

21		printf( "부산 최저기온은 몇 도인가요: " );							 

		// 4도를 입력한다고 가정 

22		scanf( "%d", &lowBusan );							 

23		printf( "제주 최저기온은 몇 도인가요: " );							 

		// 4도를 입력한다고 가정 

24		scanf( "%d", &lowJeju );								 

	 

		// 지역별 최고 기온 정보를 입력  

		// 서울, 춘천, 대전, 광주, 부산, 제주의 최고 기온을 입력 

25		printf( "\n서울 최고기온은 몇 도인가요: " );							 

		// 1도를 입력한다고 가정 

26		scanf( "%d", &highSeoul );							 

27		printf( "춘천 최고기온은 몇 도인가요: " );							 

		// 3도를 입력한다고 가정 

28		scanf( "%d", &highChuncheon );							 

29		printf( "대전 최고기온은 몇 도인가요: " );							 

		// 2도를 입력한다고 가정		 

30		scanf( "%d", &highDaejeon );							 

31		printf( "광주 최고기온은 몇 도인가요: " );							 

		// 2도를 입력한다고 가정 

32		scanf( "%d", &highGwangju ); 							 

33		printf( "부산 최고기온은 몇 도인가요: " );							 

		// 4도를 입력한다고 가정 

34		scanf( "%d", &highBusan );							 

35		printf( "제주 최고기온은 몇 도인가요: " );							 

		// 5도를 입력한다고 가정 

36		scanf( "%d", &highJeju );							 

		 

		// 날씨 정보를 바탕으로 기상 뉴스를 자동으로 생성

			  highSeoul, highChuncheon, highDaejeon, highGwangju, highBusan, highJeju ); 

40		printf( "\n일부 지방에 %s가 발효중입니다. 출근길 %s 운전에 유의해야 합니다.", weatherReport, precaution );  

41		printf( "\n이상 날씨였습니다.\n웨더봇 기자" );										 

	 

		// 프로그램 종료 

42		return 0; 

	} 

		
		

	
	

	

		
키워드 설명

		

include 
  헤더파일(.h)을 소스코드(.c)에 포함하는 명령문


stdio.h 
  standard input output의 줄임말로 printf나 scanf 등 표준입출력 관련 함수 정보를 가지고 있는 헤더파일


printf 
  print formatted의 줄임말로 전달받은 내용을 형식에 맞게 모니터에 출력하는 함수


scanf 
  scan formatted의 줄임말로 키보드에서 형식에 맞게 입력받은 값을 가져오는 함수


%s 
  문자열  형식으로 입출력


%d 
  정수 형식으로 입출력
