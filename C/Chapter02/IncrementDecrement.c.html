문제

증가연산자(++)와 감소연산자(--)의 전위연산자와 후위연산자를 실습하고 그 결과를 출력하는 프로그램입니다 이를 해결하는 다음 프로그램을 해석하세요

입출력 예제

a = 3일 때 c = ++a 적용 후 결과 a = 4, c = 4
b = 3일 때 d = b++ 적용 후 결과 b = 4, d = 3
a = 4일 때 c = --a 적용 후 결과 a = 3, c = 3
b = 4일 때 d = b-- 적용 후 결과 b = 3, d = 4

프로그램 코드 

	// 파일명 : ./Chapter02/IncrementDecrement.c 

	#include <stdio.h> 

	 

	// 프로그램 시작 

1	int main() 

	{ 

		// 변수 선언 및 초기화 

2		int a = 3, b = 3, c = 0, d = 0; 

	 

		// 증가연산자 ++의 전위연산자와 후위연산자 실습  

3		c =  

4		    ++a;									 

5		d =   

6		    b++;									 

	 

		// 증가연산자의 실습결과 출력  

7		printf( "a = 3일 때 c = ++a 적용 후 결과 a = %d, c = %d\n", a, c );		 

8		printf( "b = 3일 때 d = b++ 적용 후 결과 b = %d, d = %d\n", b, d );		 

	 

		// 감소연산자 --의 전위연산자와 후위연산자 실습 

9		c =  

10		    --a;									 

11		d =  

12		    b--;									 

	 

		// 감소연산자의 실습결과 출력  

13		printf( "a = 4일 때 c = --a 적용 후 결과 a = %d, c = %d\n", a, c );		 

14		printf( "b = 4일 때 d = b-- 적용 후 결과 b = %d, d = %d\n", b, d );		 

	 

		// 프로그램 종료 

15		return 0; 

	} 

		
		

	
	

	

		
키워드 설명

		

++a 
  a = a + 1을 먼저 실행하고 c = a을 실행하는 전위 증가 연산자


b++ 
  d = b을 먼저 실행하고 b = b + 1을 실행하는 후위 증가 연산자


--a 
  a = a - 1을 먼저 실행하고 c = a을 실행하는 전위 감소 연산자


b-- 
  d = b을 먼저 실행하고 b = b - 1을 실행하는 후위 감소 연산자
